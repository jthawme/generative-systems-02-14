import { CentreCols } from "../components/Common/CentreCols";
import { RandomGrid } from "../components/RandomGrid";

<CentreCols>
  <RandomGrid
    rows={200}
    cols={200}
    perlin
    perlinFactor={100}
    renderCell={(ctx, size, r) => {
      if (r > 0.75) {
        ctx.fillStyle = `rgb(${r * 255}, 0, 0)`;
      } else if (r > 0.5) {
        ctx.fillStyle = `rgb(0, ${r * 255}, 0)`;
      } else {
        ctx.fillStyle = `rgb(0, 0, ${r * 255})`;
      }
      ctx.fillRect(0, 0, size, size);
    }}
  />
</CentreCols>
