import { CentreCols } from "../components/Common/CentreCols";
import { RandomGrid } from "../components/RandomGrid";
import { clamp, mapRange } from "../utils/utils";

<CentreCols>
  <RandomGrid
    rows={4}
    cols={4}
    renderCell={(ctx, size, r, lastRandom) => {
      ctx.save();
      ctx.beginPath();
      const num = Math.floor(r * 6);
      switch (num) {
        case 0:
          ctx.arc(size/2,size/2,(size/2) - 1,0,Math.PI * 2);
          break;
        case 1:
          ctx.moveTo(0, size/2);
          ctx.lineTo(size/2, 0);
          ctx.moveTo(0, size);
          ctx.lineTo(size, 0);
          ctx.moveTo(size/2, size);
          ctx.lineTo(size, size/2);
          break;
        case 2:
          ctx.moveTo(0, size/2);
          ctx.lineTo(size/2, size);
          ctx.moveTo(0, 0);
          ctx.lineTo(size, size);
          ctx.moveTo(size/2, 0);
          ctx.lineTo(size, size/2);
          break;
        case 3:
          ctx.rect(1,1,size-2,size-2);
          break;
        case 4:
          ctx.moveTo(0, size/2);
          ctx.lineTo(size,size/2);
          break;
        case 5:
          ctx.moveTo(size/2, 0);
          ctx.lineTo(size/2,size);
          break;
      }
      ctx.lineWidth=2;
      ctx.stroke();
      ctx.restore();
    }}/>

</CentreCols>
